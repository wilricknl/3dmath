\subsection{Exercise 3}

\begin{enumerate}[a.]
	\item $\frac{x(1)-x(0)}{1-1} = 1$
	\item $\frac{x(2)-x(1)}{2-1} = -1$
	\item $\frac{x(2)-x(0)}{2-0} = 0$
	\item $\frac{x(6.5)-x(5.5)}{6.5-5.5} = 2$
	\item $\frac{x(9)-x(0)}{9-0} = \frac{2}{9}$
\end{enumerate}
